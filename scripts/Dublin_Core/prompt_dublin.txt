#################################################################
#    SWRL Rules (Semantic Web Rule Language)
#################################################################

# Rule 1: Documents with expired embargo become public access
# Document(?d) ^ embargoEndDate(?d, ?date) ^ swrlb:lessThan(?date, ?today) -> hasAccessRights(?d, PublicAccess)

# Rule 2: Documents without copyright holder after 70 years become public domain
# Document(?d) ^ copyrightYear(?d, ?year) ^ swrlb:subtract(?currentYear, ?year, ?age) ^ swrlb:greaterThan(?age, 70) ^ NOT hasCopyrightHolder(?d, ?holder) -> isPublicDomain(?d, true)

# Rule 3: Derived documents inherit copyright holder from source
# Document(?derived) ^ derivedFrom(?derived, ?source) ^ hasCopyrightHolder(?source, ?holder) -> hasCopyrightHolder(?derived, ?holder)

:DocumentShape rdf:type sh:NodeShape ;
    sh:targetClass :Document ;
    sh:property [
        sh:path :title ;
        sh:minCount 1 ;
        sh:datatype xsd:string ;
        sh:message "Document must have a title"
    ] ;
    sh:property [
        sh:path :creator ;
        sh:minCount 1 ;
        sh:message "Document must have at least one creator"
    ] ;
    sh:property [
        sh:path :date ;
        sh:minCount 1 ;
        sh:datatype xsd:dateTime ;
        sh:message "Document must have a creation date"
    ] ;
    sh:property [
        sh:path :rights ;
        sh:minCount 1 ;
        sh:message "Document must have rights information"
    ] .

:CopyrightDocumentShape rdf:type sh:NodeShape ;
    sh:targetClass :Document ;
    sh:property [
        sh:path :hasCopyrightHolder ;
        sh:minCount 1 ;
        sh:message "Copyrighted document must have a copyright holder"
    ] ;
    sh:property [
        sh:path :copyrightYear ;
        sh:datatype xsd:gYear ;
        sh:message "Copyright year must be a valid year"
    ] .

    :title "Annual Report 2024" ;
    :creator "John Doe" ;
    :date "2024-12-15T10:30:00"^^xsd:dateTime ;
    :rights "Â© 2024 ACME Corporation. All rights reserved." ;
    :copyrightYear "2024"^^xsd:gYear ;
    :copyrightNotice "This document is proprietary and confidential" ;
    :hasCopyrightHolder :acme_corp ;
    :hasOwner :acme_corp ;
    :hasLicense :AllRightsReserved ;
    :hasAccessRights :RestrictedAccess ;
    :hasMetadataExtraction :extraction_001 .

:acme_corp rdf:type :Organization ;
    foaf:name "ACME Corporation" ;
    rdfs:label "ACME Corporation" .

:extraction_001 rdf:type :AutomaticExtraction ;
    :extractionConfidence "0.95"^^xsd:decimal ;
    prov:generatedAtTime "2025-01-02T14:30:00"^^xsd:dateTime .

# Example XLS Document with shared ownership
:example_xls_001 rdf:type :XLSDocument ;
